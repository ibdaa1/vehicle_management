What SHOULD exist in app.js:

JavaScript
// Apply data-label-ar/data-label-en translations to all elements
function applyDataLabelTranslations() {
    const lang = i18n.lang || 'ar';
    const attr = lang === 'ar' ? 'data-label-ar' : 'data-label-en';
    
    document.querySelectorAll('[data-label-ar][data-label-en]').forEach(el => {
        const text = el.getAttribute(attr);
        if (text && !el.textContent) {
            el.textContent = text;
        }
    });
}
